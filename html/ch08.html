요소 여러 개에 접근하기

  • document.getElementsByClass("class")

태그 선택자로 요소에 접근하기

  • document.querySelectorAll("s"): s 태그 선택자를 사용하여 요소에 접

  • .b - class b
  • c -> tag c

체크박스 값 여러 개 읽기

function getCheckBox() {
  let sum = 0;  const adds = document.getElementsByName("myAdd");  for (let i = 0; i < adds.length; i++) {
    if (adds[i].checked) {
      sum = sum + parseInt(adds[i].value);    }
  }
  document.getElementById("result").innerHTML = `추가 금액: ${sum}원`}

라디오 버튼에서 값 읽기

function getRadio() {
  const choice = document.querySelectorAll("[type='radio']");  for (let i = 0; i < choice.length; i++) {
    if (choice[i].checked) {
      document.getElementById("result").innerHTML = `${choice[i].value}`;      break;    }
  }
}

드롭박스에서 값 읽기

function getSelect() {
  const index = document.getElementById("myOption").selectedIndex;  const st = document.getElementById("myOption").options[index].value;  document.getElementById("result").innerHTML = st;}

HTML 요소 변경하기

내용 변경하기

  • 텍스트 출력 메서드
    • document.write(): html페이지를 새로 만들고 글자를 출력
    • document.writeln(): 글자 끝에 줄바꿈 문자를 추가하여 출력
  • 문자 노드에 접근하는 속성
    • innerHTML: HTML 태그 요소를 직접 변경, 태그 인식 가능
    • innerText: HTML이 적용된 상태로 불러옴. display: none은 불러올 수 없음
    • textContent: HTML을 제외한 모든 문자를 불러옴. display: none의 문자도 가져옴
